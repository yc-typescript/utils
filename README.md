```bash
npm i -S @yct/utils
```

## Functions

### fetch
> browser only
```ts
function fetch(
  method: string,
  url: string,
  params: any = {},
  headers: any = {},
  json: boolean = false,
  responseType: string = ''
): Promise<any>
```
> Usage
```ts
import { fetch } from "@yct/utils";

const res = await fetch('GET', 'http://localhost');
```
